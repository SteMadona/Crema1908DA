Premi tesseramento

Under 17 – Stagione 2025/26

Note

Regola: il premio di tesseramento è dovuto per ogni stagione dal 20/21 in cui il giocatore risulta in una società diversa da AC Crema 1908. Le celle vuote dal 20/21 in poi vengono marcate come dati mancanti.

library(dplyr) library(stringr) library(readr) library(tidyr)

=== CONFIG ===

datafile <- read_excel(here(“data”, “STORICO_CALCIATORI.xlsx”)) crema_tag <- “AC CREMA 1908” due_from <- “20/21” # da qui in avanti si conteggiano i premi

— Lettura dati —

df <- read_csv(datafile, show_col_types = FALSE)

Elimina eventuali colonne completamente vuote (capita con export Excel)

df <- df %>% select(where(~ !all(is.na(.) | str_squish(as.character(.)) == ““)))

Individua le colonne stagione (formato 20/21)

season_cols <- names(df)[str_detect(names(df), “^\d{2}/\d{2}$”)] if (length(season_cols) == 0) stop(“Non trovo colonne stagione nel formato ‘20/21’.”)

Ordine stagioni come appaiono nel file (se vuoi forzarlo, sostituisci season_cols con un vettore fisso)

seasons <- season_cols if (!due_from %in% seasons) stop(paste(“La stagione”, due_from, “non è tra le colonne.”))

due_seasons <- seasons[match(due_from, seasons):length(seasons)]

norm <- function(x) str_squish(replace_na(as.character(x), ““)) is_crema <- function(x) str_detect(str_to_upper(norm(x)), str_to_upper(crema_tag))

— Calcoli per riga —

calc_arrival <- function(row) { idx <- which(is_crema(row[seasons])) if (length(idx) == 0) return(“—”) seasons[min(idx)] }

calc_due_missing <- function(row) { vals <- setNames(lapply(row[due_seasons], norm), due_seasons) missing <- names(vals)[unlist(vals) == “”] due <- names(vals)[unlist(vals) != “” & !sapply(vals, is_crema)] list( due_years = if (length(due)) paste(due, collapse = “,”) else “—”, due_n = length(due), missing = if (length(missing)) paste(missing, collapse = “,”) else “—” ) }

tmp <- df %>% rowwise() %>% mutate( Anno arrivo (Crema) = calc_arrival(cur_data()), .calc = list(calc_due_missing(cur_data())), Anni premio (da 20/21) = .calc\(due_years, `N° anni premio` = .calc\)due_n, Dati mancanti (da 20/21) = .calc$missing ) %>% ungroup() %>% select(-.calc)

— KPI —

n_players <- nrow(tmp) n_due_players <- sum(tmp\(`N° anni premio` > 0, na.rm = TRUE) tot_due_years <- sum(tmp\)N° anni premio, na.rm = TRUE) n_missing <- sum(tmp$Dati mancanti (da 20/21) != “—”, na.rm = TRUE)

Value boxes (stile professionale)

library(bslib) bslib::layout_column_wrap( width = 1/4, bslib::value_box(“Giocatori”, n_players), bslib::value_box(“Con premio dovuto”, n_due_players), bslib::value_box(“Anni premio totali”, tot_due_years), bslib::value_box(“Righe con dati mancanti”, n_missing) )

library(DT)

view <- tmp %>% select(Giocatore, Categoria, Nascita, Anno arrivo (Crema), N° anni premio, Anni premio (da 20/21), Dati mancanti (da 20/21)) %>% arrange(desc(N° anni premio), Giocatore)

dt <- datatable( view, rownames = FALSE, extensions = c(“Buttons”), options = list( dom = “Bfrtip”, buttons = c(“copy”, “csv”, “excel”, “print”), pageLength = 25, scrollX = TRUE ) )

Evidenzia chi ha premio e chi ha dati mancanti

dt %>% formatStyle( “N° anni premio”, backgroundColor = styleInterval(0, c(NA, “#fff3cd”)), fontWeight = styleInterval(0, c(“normal”, “bold”)) ) %>% formatStyle( “Dati mancanti (da 20/21)”, backgroundColor = styleEqual(“—”, NA, default = “#f8d7da”) )

storico <- tmp %>% select(Giocatore, all_of(seasons)) %>% arrange(Giocatore)

datatable(storico, rownames = FALSE, options = list(pageLength = 25, scrollX = TRUE))
